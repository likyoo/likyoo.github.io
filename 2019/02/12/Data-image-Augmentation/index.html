Data(image) Augmentation

2019-02-12 machine learning 0 评论 字数统计: 285(字) 阅读时长: 1(分)

Data Augmentation 是 Regularization 的一种方法

 

可以做水平翻转

可以抽取不同尺度大小的裁剪图像，在测试时，评估一些固定的裁剪图像

可以做色彩抖动：

还可以：

  1. import os
  2. import random
  3. from PIL import Image
  4. from PIL import ImageEnhance
  5.  
  6. def enhance_image(image_path):
  7.     
  8.     image = Image.open(image_path)
  9.     
  10.     # 随机亮度
  11.     enhance_image = ImageEnhance.Brightness(image)
  12.     bright = random.uniform(0.2, 1.8)
  13.     enhance_image = enhance_image.enhance(bright)
  14.     
  15.     # 随机饱和度
  16.     enhance_image = ImageEnhance.Contrast(enhance_image)
  17.     contrast = random.uniform(0.2, 1.8)
  18.     enhance_image = enhance_image.enhance(contrast)
  19.     
  20.     # 随机左右翻转
  21.     if random.randint(0,1) == 1:
  22.         enhance_image = enhance_image.transpose(Image.FLIP_LEFT_RIGHT)
  23.         
  24.     return enhance_image
  25.  
  26. def add_image(folder_list, folder_path, max_number):
  27.     
  28.     folder_length = len(folder_list)
  29.     random_folder_list = []
  30.     
  31.     for _ in range(max_number - folder_length):  
  32.         index = random.randint(0, folder_length - 1)
  33.         random_folder_list.append(folder_list[index])
  34.     
  35.     add_path = "add"
  36.     full_add_path = folder_path + os.sep + add_path
  37.     if not os.path.exists(full_add_path):
  38.         os.mkdir(full_add_path)
  39.     
  40.     counter = 0
  41.     for path in random_folder_list:
  42.         new_image = enhance_image(path)
  43.         new_image.save(full_add_path + os.sep + str(counter) + ".jpg")
  44.         counter += 1
  45.  
  46. def image_augmentation(db_folder, limit_number=100, max_number=600):
  47.     
  48.     number = 0
  49.     for people_folder in os.listdir(db_folder):
  50.         number += 1
  51.         
  52.         src_people_path = db_folder + os.sep + people_folder
  53.         
  54.         counter = 0
  55.         people_folder_list = []
  56.         for vedio_folder in os.listdir(src_people_path):
  57.             
  58.             src_vedio_path = src_people_path + os.sep + vedio_folder
  59.  
  60.             for img_file in os.listdir(src_vedio_path):
  61.                 counter += 1
  62.                 src_img_path = src_vedio_path + os.sep + img_file
  63.                 people_folder_list.append(src_img_path)
  64.         
  65.         if counter < limit_number or counter >= max_number:
  66.             continue
  67.         else:
  68.             add_image(people_folder_list, src_people_path, max_number)
  69.             
  70.             print(people_folder + ": id--->" + str(number) + "--->OK")
  71.  
  72. def run():
  73.     random.seed(7)
  74.     db_folder = "image\\origin"
  75.     image_augmentation(db_folder)
  76.  
  77. if __name__ == "__main__":
  78.     run()
