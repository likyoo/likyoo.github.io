  2.  
  3.  
  4.     def __init__(self, input_dim=3*32*32, hidden_dim=50, num_classes=10,
  5.                  weight_scale=1e-3, reg=0.8):
  6.  
  7.         self.params = {}
  8.         self.reg = reg
  9.  
  10.         ############################################################################
  11.         # TODO: Initialize the weights and biases of the two-layer net. Weights    #
  12.         # should be initialized from a Gaussian centered at 0.0 with               #
  13.         # standard deviation equal to weight_scale, and biases should be           #
  14.         # initialized to zero. All weights and biases should be stored in the      #
  15.         # dictionary self.params, with first layer weights                         #
  16.         # and biases using the keys 'W1' and 'b1' and second layer                 #
  17.         # weights and biases using the keys 'W2' and 'b2'.                         #
  18.         ############################################################################
  19.         self.params['W1'] = weight_scale * np.random.randn(input_dim, hidden_dim)
  20.         self.params['b1'] = np.zeros(hidden_dim)
  21.         self.params['W2'] = weight_scale * np.random.randn(hidden_dim, num_classes)
  22.         self.params['b2'] = np.zeros(num_classes)
  23.         ############################################################################
  24.         #                             END OF YOUR CODE                             #
  25.         ############################################################################
  26.  
  27.  
  28.     def loss(self, X, y=None):
  29.         """
  30.         Compute loss and gradient for a minibatch of data.
  32.         Inputs:
  33.         - X: Array of input data of shape (N, d_1, ..., d_k)
  34.         - y: Array of labels, of shape (N,). y[i] gives the label for X[i].
  36.         Returns:
  37.         If y is None, then run a test-time forward pass of the model and return:
  38.         - scores: Array of shape (N, C) giving classification scores, where
  39.           scores[i, c] is the classification score for X[i] and class c.
  41.         If y is not None, then run a training-time forward and backward pass and
  42.         return a tuple of:
  43.         - loss: Scalar value giving the loss
  44.         - grads: Dictionary with the same keys as self.params, mapping parameter
  45.           names to gradients of the loss with respect to those parameters.
  46.         """
  47.         scores = None
  48.         ############################################################################
  49.         # TODO: Implement the forward pass for the two-layer net, computing the    #
  50.         # class scores for X and storing them in the scores variable.              #
  51.         ############################################################################
  52.         # Unpack variables from the params dictionary
  53.         W1, b1 = self.params['W1'], self.params['b1']
  54.         W2, b2 = self.params['W2'], self.params['b2']
  55.         X = X.reshape(X.shape[0], -1)
  56.  
  57.         # 计算分值
  58.         h1_out,h1_cache = affine_relu_forward(X, W1, b1)
  59.         scores,out_cache = affine_forward(h1_out, W2, b2)
  60.         #############################################################################
  61.         #                              END OF YOUR CODE                             #
  62.         #############################################################################
  63.  
  64.         # If the targets are not given then jump out, we're done
  65.         if y is None:
  66.             return scores
  67.  
  68.         # Compute the loss
  69.         loss = None
  70.         #############################################################################
  71.         # TODO: Finish the forward pass, and compute the loss. This should include  #
  72.         # both the data loss and L2 regularization for W1 and W2. Store the result  #
  73.         # in the variable loss, which should be a scalar. Use the Softmax           #
  74.         # classifier loss.                                                          #
  75.         #############################################################################
  76.         # 最后一层softmax
  77.         loss,dout = softmax_loss(scores,y)
  78.         loss += 0.5 * self.reg * (np.sum(W1 ** 2) + np.sum(W2 ** 2))
  79.         #############################################################################
  80.         #                              END OF YOUR CODE                             #
  81.         #############################################################################
  82.  
  83.         # Backward pass: compute gradients
  84.         grads = {}
  85.         #############################################################################
  86.         # TODO: Compute the backward pass, computing the derivatives of the weights #
  87.         # and biases. Store the results in the grads dictionary. For example,       #
  88.         # grads['W1'] should store the gradient on W1, and be a matrix of same size #
  89.         #############################################################################
  90.         # 计算梯度
  91.         dout, dw2, db2 = affine_backward(dout, out_cache)
  92.         dw2 += self.reg * W2
  93.         grads['W2'] = dw2
  94.         grads['b2'] = db2
  95.         _, dw1, db1 = affine_relu_backward(dout, h1_cache)
  96.         dw1 += self.reg * W1
  97.         grads['W1'] = dw1
  98.         grads['b1'] = db1
  99.         #############################################################################
  100.         #                              END OF YOUR CODE                             #
  101.         #############################################################################
  102.  
  103.         return loss, grads

  2.  
  3.  
  4.     def __init__(self, hidden_dims, input_dim=3*32*32, num_classes=10,
  5.                  dropout=1, normalization=None, reg=0.0,
  6.                  weight_scale=1e-2, dtype=np.float32, seed=None):
  7.  
  8.         self.normalization = normalization
  9.         self.use_dropout = dropout != 1
  10.         self.reg = reg
  11.         self.num_layers = 1 + len(hidden_dims)
  12.         self.dtype = dtype
  13.         self.params = {}
  14.  
  15.         ############################################################################
  16.         # TODO: Initialize the parameters of the network, storing all values in    #
  17.         # the self.params dictionary. Store weights and biases for the first layer #
  18.         # in W1 and b1; for the second layer use W2 and b2, etc. Weights should be #
  19.         # initialized from a normal distribution centered at 0 with standard       #
  20.         # deviation equal to weight_scale. Biases should be initialized to zero.   #
  21.         #                                                                          #
  22.         # When using batch normalization, store scale and shift parameters for the #
  23.         # first layer in gamma1 and beta1; for the second layer use gamma2 and     #
  24.         # beta2, etc. Scale parameters should be initialized to ones and shift     #
  25.         # parameters should be initialized to zeros.                               #
  26.         ############################################################################
  27.         in_dim = input_dim
  28.         for i, h_dim in enumerate(hidden_dims):
  29.             self.params['W%d' %(i+1,)] = weight_scale * np.random.randn(in_dim, h_dim)
  30.             self.params['b%d' %(i+1,)] = np.zeros((h_dim,))
  31.             if self.normalization=='batchnorm':
  32.                 self.params['gamma%d' %(i+1,)] = np.ones((h_dim,))
  33.                 self.params['bets%d' %(i+1,)] = np.zeros((h_dim,))
  34.             in_dim = h_dim
  35.         self.params['W%d' %(self.num_layers,)] = weight_scale * np.random.randn(in_dim, num_classes)
  36.         self.params['b%d' %(self.num_layers,)] = np.zeros((num_classes,))
  37.         ############################################################################
  38.         #                             END OF YOUR CODE                             #
  39.         ############################################################################
  40.  
  41.         # When using dropout we need to pass a dropout_param dictionary to each
  42.         # dropout layer so that the layer knows the dropout probability and the mode
  43.         # (train / test). You can pass the same dropout_param to each dropout layer.
  44.         self.dropout_param = {}
  45.         if self.use_dropout:
  46.             self.dropout_param = {'mode': 'train', 'p': dropout}
  47.             if seed is not None:
  48.                 self.dropout_param['seed'] = seed
  49.  
  50.         # With batch normalization we need to keep track of running means and
  51.         # variances, so we need to pass a special bn_param object to each batch
  52.         # normalization layer. You should pass self.bn_params[0] to the forward pass
  53.         # of the first batch normalization layer, self.bn_params[1] to the forward
  54.         # pass of the second batch normalization layer, etc.
  55.         self.bn_params = []
  56.         if self.normalization=='batchnorm':
  57.             self.bn_params = [{'mode': 'train'} for i in range(self.num_layers - 1)]
  58.         if self.normalization=='layernorm':
  59.             self.bn_params = [{} for i in range(self.num_layers - 1)]
  60.  
  61.         # Cast all parameters to the correct datatype
  62.         for k, v in self.params.items():
  63.             self.params[k] = v.astype(dtype)
  64.  
  65.  
  66.     def loss(self, X, y=None):
  67.         """
  68.         Compute loss and gradient for the fully-connected net.
  70.         Input / output: Same as TwoLayerNet above.
  71.         """
  72.         X = X.astype(self.dtype)
  73.         mode = 'test' if y is None else 'train'
  74.
