cs231n assignment2 Batch Normalization

2019-02-09 cs231n 0 评论 字数统计: 2k(字) 阅读时长: 12(分)

Batch normalization 优点：

  • 减少坏初始化的影响
  • 加快模型的收敛速度
  • 可以用大些的学习率
  • 能有效地防止过拟合

 batchnorm_forward 实现：

  1. def batchnorm_forward(x, gamma, beta, bn_param):
  2.     
  3.     mode = bn_param['mode']
  4.     eps = bn_param.get('eps', 1e-5)
  5.     momentum = bn_param.get('momentum', 0.9)
  6.  
  7.     N, D = x.shape
  8.     running_mean = bn_param.get('running_mean', np.zeros(D, dtype=x.dtype))
  9.     running_var = bn_param.get('running_var', np.zeros(D, dtype=x.dtype))
  10.  
  11.     out, cache = None, None
  12.     if mode == 'train':
  13.         #######################################################################
  14.         # TODO: Implement the training-time forward pass for batch norm.      #
  15.         # Use minibatch statistics to compute the mean and variance, use      #
  16.         # these statistics to normalize the incoming data, and scale and      #
  17.         # shift the normalized data using gamma and beta.                     #
  18.         #                                                                     #
  19.         # You should store the output in the variable out. Any intermediates  #
  20.         # that you need for the backward pass should be stored in the cache   #
  21.         # variable.                                                           #
  22.         #                                                                     #
  23.         # You should also use your computed sample mean and variance together #
  24.         # with the momentum variable to update the running mean and running   #
  25.         # variance, storing your result in the running_mean and running_var   #
  26.         # variables.                                                          #
  27.         #                                                                     #
  28.         # Note that though you should be keeping track of the running         #
  29.         # variance, you should normalize the data based on the standard       #
  30.         # deviation (square root of variance) instead!                        # 
  31.         # Referencing the original paper (https://arxiv.org/abs/1502.03167)   #
  32.         # might prove to be helpful.                                          #
  33.         #######################################################################
  34.         sample_mean = np.mean(x, axis=0)
  35.         sample_var = np.var(x, axis=0)
  36.         x_hat = (x - sample_mean) / (np.sqrt(sample_var + eps))
  37.         out = gamma * x_hat + beta
  38.  
  39.         cache = (x, sample_mean, sample_var, x_hat, eps, gamma, beta)
  40.         running_mean = momentum * running_mean + (1 - momentum) * sample_mean
  41.         running_var = momentum * running_var + (1 - momentum) * sample_var
  42.         #######################################################################
  43.         #                           END OF YOUR CODE                          #
  44.         #######################################################################
  45.     elif mode == 'test':
  46.         #######################################################################
  47.         # TODO: Implement the test-time forward pass for batch normalization. #
  48.         # Use the running mean and variance to normalize the incoming data,   #
  49.         # then scale and shift the normalized data using gamma and beta.      #
  50.         # Store the result in the out variable.                               #
  51.         #######################################################################
  52.         out = (x - running_mean) * gamma / (np.sqrt(running_var))
  53.         #######################################################################
  54.         #                          END OF YOUR CODE                           #
  55.         #######################################################################
  56.     else:
  57.         raise ValueError('Invalid forward batchnorm mode "%s"' % mode)
  58.  
  59.     # Store the updated running means back into bn_param
  60.     bn_param['running_mean'] = running_mean
  61.     bn_param['running_var'] = running_var
  62.  
  63.     return out, cache

 

Batch normalization: backward 

  1. # Batch normalization: backward
  2. # Gradient check batchnorm backward pass
  3. np.random.seed(231)
  4. N, D = 4, 5
  5. x = 5 * np.random.randn(N, D) + 12
  6. gamma = np.random.randn(D)
  7. beta = np.random.randn(D)
  8. dout = np.random.randn(N, D)
  9.  
  10. bn_param = {'mode': 'train'}
  11. fx = lambda x: batchnorm_forward(x, gamma, beta, bn_param)[0]
  12. fg = lambda a: batchnorm_forward(x, a, beta, bn_param)[0]
  13. fb = lambda b: batchnorm_forward(x, gamma, b, bn_param)[0]
  14.  
  15. dx_num = eval_numerical_gradient_array(fx, x, dout)
  16. da_num = eval_numerical_gradient_array(fg, gamma.copy(), dout)
  17. db_num = eval_numerical_gradient_array(fb, beta.copy(), dout)
  18.  
  19. _, cache = batchnorm_forward(x, gamma, beta, bn_param)
  20. dx, dgamma, dbeta = batchnorm_backward(dout, cache)
  21. #You should expect to see relative errors between 1e-13 and 1e-8
  22. print('dx error: ', rel_error(dx_num, dx))
  23. print('dgamma error: ', rel_error(da_num, dgamma))
  24. print('dbeta error: ', rel_error(db_num, dbeta))

batchnorm_backward 实现： 

  1. def batchnorm_backward(dout, cache):
  2.  
  3.     dx, dgamma, dbeta = None, None, None
  4.     ###########################################################################
  5.     # TODO: Implement the backward pass for batch normalization. Store the    #
  6.     # results in the dx, dgamma, and dbeta variables.                         #
  7.     # Referencing the original paper (https://arxiv.org/abs/1502.03167)       #
  8.     # might prove to be helpful.                                              #
  9.     ###########################################################################
  10.     x, mean, var, x_hat, eps, gamma, beta = cache
  11.     N = x.shape[0]
  12.     dgamma = np.sum(dout * x_hat, axis=0)
  13.     dbeta = np.sum(dout * 1.0, axis=0)
  14.     dx_hat = dout * gamma
  15.     dx_hat_numerator = dx_hat / np.sqrt(var + eps)
  16.     dx_hat_denominator = np.sum(dx_hat * (x - mean), axis=0)
  17.     dx_1 = dx_hat_numerator
  18.     dvar = -0.5 * ((var + eps) ** (-1.5)) * dx_hat_denominator
  19.     dmean = -1.0 * np.sum(dx_hat_numerator, axis=0) + dvar * np.mean(-2.0 * (x - mean),axis=0)
  20.     dx_var = dvar * 2.0 / N * (x - mean)
  21.     dx_mean = dmean * 1.0 / N
  22.     dx = dx_1 + dx_var + dx_mean
  23.     ###########################################################################
  24.     #                             END OF YOUR CODE                            #
  25.     ###########################################################################
  26.  
  27.     return dx, dgamma, dbeta

Batch normalization and batch size

  1. # Batch normalization and batch size
  2. def run_batchsize_experiments(normalization_mode):
