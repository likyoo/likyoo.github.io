Dropout forward pass

  1. np.random.seed(231)
  2. x = np.random.randn(500, 500) + 10
  3.  
  4. for p in [0.25, 0.4, 0.7]:
  5.   out, _ = dropout_forward(x, {'mode': 'train', 'p': p})
  6.   out_test, _ = dropout_forward(x, {'mode': 'test', 'p': p})
  7.  
  8.   print('Running tests with p = ', p)
  9.   print('Mean of input: ', x.mean())
  10.   print('Mean of train-time output: ', out.mean())
  11.   print('Mean of test-time output: ', out_test.mean())
  12.   print('Fraction of train-time output set to zero: ', (out == 0).mean())
  13.   print('Fraction of test-time output set to zero: ', (out_test == 0).mean())
  14.   print()

dropout_forward 实现：

    这里在training中多除以一个keep_prob，在testing中就不用做任何操作了

  1. def dropout_forward(x, dropout_param):
  2.    
  3.     p, mode = dropout_param['p'], dropout_param['mode']
  4.     if 'seed' in dropout_param:
  5.         np.random.seed(dropout_param['seed'])
  6.  
  7.     mask = None
  8.     out = None
  9.  
  10.     if mode == 'train':
  11.         #######################################################################
  12.         # TODO: Implement training phase forward pass for inverted dropout.   #
  13.         # Store the dropout mask in the mask variable.                        #
  14.         #######################################################################
  15.         keep_prob = 1 - p
  16.         mask = (np.random.rand(*x.shape) < keep_prob) / keep_prob
  17.         out = mask * x
  18.         #######################################################################
  19.         #                           END OF YOUR CODE                          #
  20.         #######################################################################
  21.     elif mode == 'test':
  22.         #######################################################################
  23.         # TODO: Implement the test phase forward pass for inverted dropout.   #
  24.         #######################################################################
  25.         out = x
  26.         #######################################################################
  27.         #                            END OF YOUR CODE                         #
  28.         #######################################################################
  29.  
  30.     cache = (dropout_param, mask)
  31.     out = out.astype(x.dtype, copy=False)
  32.  
  33.     return out, cache

 

Dropout backward pass

  1. np.random.seed(231)
  2. x = np.random.randn(10, 10) + 10
  3. dout = np.random.randn(*x.shape)
  4.  
  5. dropout_param = {'mode': 'train', 'p': 0.2, 'seed': 123}
  6. out, cache = dropout_forward(x, dropout_param)
  7. dx = dropout_backward(dout, cache)
  8. dx_num = eval_numerical_gradient_array(lambda xx: dropout_forward(xx, dropout_param)[0], x, dout)
  9.  
  10. # Error should be around e-10 or less
  11. print('dx relative error: ', rel_error(dx, dx_num))

dropout_backward 实现：

  1. def dropout_backward(dout, cache):
  2.  
  3.     dropout_param, mask = cache
  4.     mode = dropout_param['mode']
  5.  
  6.     dx = None
  7.     if mode == 'train':
  8.         #######################################################################
  9.         # TODO: Implement training phase backward pass for inverted dropout   #
  10.         #######################################################################
  11.         dx = mask * dout
  12.         #######################################################################
  13.         #                          END OF YOUR CODE                           #
  14.         #######################################################################
  15.     elif mode == 'test':
  16.         dx = dout
  17.     return dx
