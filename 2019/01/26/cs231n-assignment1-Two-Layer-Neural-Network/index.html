在nerual_net.py 的 loss 中： 求导过程在注释中

  1. # Backward pass: compute gradients
  2.     grads = {}
  3.     #############################################################################
  4.     # TODO: Compute the backward pass, computing the derivatives of the weights #
  5.     # and biases. Store the results in the grads dictionary. For example,       #
  6.     # grads['W1'] should store the gradient on W1, and be a matrix of same size #
  7.     #############################################################################
  8.     # 计算梯度
  9.     # softmax
  10.     dscores = probs
  11.     dscores[range(N), y] -= 1
  12.     dscores /= N    #
  13.  
  14.     # W2 和 b2
  15.     grads['W2'] = np.dot(a1.T, dscores)       # 见开头推倒过程或PPT
  16.     grads['b2'] = np.sum(dscores, axis=0)     # 使其与b2的size相同
  17.     # backward 所传播的df/dx
  18.     dhidden = np.dot(dscores, W2.T)
  19.     # relu求导
  20.     dhidden[a1 <= 0] = 0
  21.  
  22.     # W1 和 b1
  23.     grads['W1'] = np.dot(X.T, dhidden)
  24.     grads['b1'] = np.sum(dhidden, axis=0)
  25.  
  26.     # 加上正则化梯度的部分
  27.     grads['W2'] += reg * W2
  28.     grads['W1'] += reg * W1
  29.     #############################################################################
  30.     #                              END OF YOUR CODE                             #
  31.     #############################################################################

 这里想要得到的梯度是各个参数的，即 df/dw df/db，backward传递的是 df/dx

这里依然是用 mini-batch

  1.   def train(self, X, y, X_val, y_val,
  2.             learning_rate=1e-3, learning_rate_decay=0.95,
  3.             reg=5e-6, num_iters=100,
  4.             batch_size=200, verbose=False):
  5.  
  6.     num_train = X.shape[0]
  7.     iterations_per_epoch = max(num_train / batch_size, 1)
  8.  
  9.     # Use SGD to optimize the parameters in self.model
  10.     loss_history = []
  11.     train_acc_history = []
  12.     val_acc_history = []
  13.   # 个人感觉一个epoch中应该batch所有的数据，这里直接用epoch应该结果也差不多
  14.     for it in range(num_iters):
  15.       X_batch = None
  16.       y_batch = None
  17.  
  18.       #########################################################################
  19.       # TODO: Create a random minibatch of training data and labels, storing  #
  20.       # them in X_batch and y_batch respectively.                             #
  21.       #########################################################################
  22.       sample_indices = np.random.choice(np.arange(num_train), batch_size)
  23.       X_batch = X[sample_indices]
  24.       y_batch = y[sample_indices]
  25.       #########################################################################
  26.       #                             END OF YOUR CODE                          #
  27.       #########################################################################
  28.  
  29.       # Compute loss and gradients using the current minibatch
  30.       loss, grads = self.loss(X_batch, y=y_batch, reg=reg)
  31.       loss_history.append(loss)
  32.  
  33.       #########################################################################
  34.       # TODO: Use the gradients in the grads dictionary to update the         #
  35.       # parameters of the network (stored in the dictionary self.params)      #
  36.       # using stochastic gradient descent. You'll need to use the gradients   #
  37.       # stored in the grads dictionary defined above.                         #
  38.       #########################################################################
  39.       # gradient desent
  40.       self.params['W1'] += -learning_rate * grads['W1']
  41.       self.params['b1'] += -learning_rate * grads['b1']
  42.       self.params['W2'] += -learning_rate * grads['W2']
  43.       self.params['b2'] += -learning_rate * grads['b2']
  44.  
  45.  
  46.       #########################################################################
  47.       #                             END OF YOUR CODE                          #
  48.       #########################################################################
  49.  
  50.       if verbose and it % 100 == 0:
  51.         print('iteration %d / %d: loss %f' % (it, num_iters, loss))
  52.  
  53.       # Every epoch, check train and val accuracy and decay learning rate.
  54.       if it % iterations_per_epoch == 0:
  55.         # Check accuracy
  56.         train_acc = (self.predict(X_batch) == y_batch).mean()
  57.         val_acc = (self.predict(X_val) == y_val).mean()
  58.         train_acc_history.append(train_acc)
  59.         val_acc_history.append(val_acc)
  60.  
  61.         # Decay learning rate
  62.         learning_rate *= learning_rate_decay
  63.  
  64.     return {
  65.       'loss_history': loss_history,
  66.       'train_acc_history': train_acc_history,
  67.       'val_acc_history': val_acc_history,
  68.     }
