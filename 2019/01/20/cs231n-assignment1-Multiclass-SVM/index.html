  1. def svm_loss_naive(W, X, y, reg):
  2.   """
  3.   Structured SVM loss function, naive implementation (with loops).
  5.   Inputs have dimension D, there are C classes, and we operate on minibatches
  6.   of N examples.
  8.   Inputs:
  9.   - W: A numpy array of shape (D, C) containing weights.
  10.   - X: A numpy array of shape (N, D) containing a minibatch of data.
  11.   - y: A numpy array of shape (N,) containing training labels; y[i] = c means
  12.     that X[i] has label c, where 0 <= c < C.
  13.   - reg: (float) regularization strength
  15.   Returns a tuple of:
  16.   - loss as single float
  17.   - gradient with respect to weights W; an array of same shape as W
  18.   """
  19.   dW = np.zeros(W.shape) # initialize the gradient as zero
  20.  
  21.   # compute the loss and the gradient
  22.   num_classes = W.shape[1]
  23.   num_train = X.shape[0]
  24.   loss = 0.0
  25.   for i in range(num_train):
  26.     scores = X[i].dot(W)
  27.     correct_class_score = scores[y[i]]
  28.     for j in range(num_classes):
  29.       if j == y[i]:
  30.         continue
  31.       margin = scores[j] - correct_class_score + 1 # note delta = 1
  32.       if margin > 0:
  33.         loss += margin
  34.         # 在loss function上分别对 W[Y[i]] 和 W[j] 求导得
  35.         dW[:,y[i]] += -X[i,:].T
  36.         dW[:,j] += X[i,:].T
  37.  
  38.   # Right now the loss is a sum over all training examples, but we want it
  39.   # to be an average instead so we divide by num_train.
  40.   loss /= num_train
  41.   dW /= num_train
  42.  
  43.   # Add regularization to the loss.
  44.   loss += reg * np.sum(W * W)
  45.   dW += reg * W
  46.  
  47.   #############################################################################
  48.   # TODO:                                                                     #
  49.   # Compute the gradient of the loss function and store it dW.                #
  50.   # Rather that first computing the loss and then computing the derivative,   #
  51.   # it may be simpler to compute the derivative at the same time that the     #
  52.   # loss is being computed. As a result you may need to modify some of the    #
  53.   # code above to compute the gradient.                                       #
  54.   #############################################################################
  55.  
  56.  
  57.   return loss, dW

  1. def grad_check_sparse(f, x, analytic_grad, num_checks=10, h=1e-5):
  2.   """
  3.   sample a few random elements and only return numerical
  4.   in this dimensions.
  5.   """
  6.  
  7.   for i in range(num_checks):
  8.     # 随机取出 W 中的一个数
  9.     ix = tuple([randrange(m) for m in x.shape])
  10.  
  11.     # 计算数值梯度
  12.     oldval = x[ix]
  13.     x[ix] = oldval + h # increment by h
  14.     fxph = f(x) # evaluate f(x + h)
  15.     x[ix] = oldval - h # increment by h
  16.     fxmh = f(x) # evaluate f(x - h)
  17.     x[ix] = oldval # reset
  18.  
  19.     # 计算数值梯度和解析梯度的差值
  20.     grad_numerical = (fxph - fxmh) / (2 * h)
  21.     grad_analytic = analytic_grad[ix]
  22.     rel_error = abs(grad_numerical - grad_analytic) / (abs(grad_numerical) + abs(grad_analytic))
  23.     print('numerical: %f analytic: %f, relative error: %e' % (grad_numerical, grad_analytic, rel_error))

  1. def svm_loss_vectorized(W, X, y, reg):
  2.   """
  3.   Structured SVM loss function, vectorized implementation.
  5.   Inputs and outputs are the same as svm_loss_naive.
  6.   """
  7.   loss = 0.0
  8.   dW = np.zeros(W.shape) # initialize the gradient as zero
  9.  
  10.   #############################################################################
  11.   # TODO:                                                                     #
  12.   # Implement a vectorized version of the structured SVM loss, storing the    #
  13.   # result in loss.                                                           #
  14.   #############################################################################
  15.   # 实现结构化SVM损失函数的向量版本
  16.   scores = X.dot(W)                           # 得到一个 num_train * 10的矩阵
  17.   num_classes = W.shape[1]
  18.   num_train = X.shape[0]
  19.  
  20.   scores_correct = scores[np.arange(num_train), y]      # 用一个list小技巧直接取出所有的 scores[y[i]]
  21.   scores_correct = np.reshape(scores_correct, (num_train, -1))
  22.  
  23.   margins = scores - scores_correct + 1
  24.   margins = np.maximum(0, margins)
  25.   margins[np.arange(num_train), y] = 0
  26.   loss += np.sum(margins) / num_train
  27.   loss += 0.5 * reg * np.sum(W * W)
  28.  
  29.   #############################################################################
  30.   #                             END OF YOUR CODE                              #
  31.   #############################################################################
  32.  
  33.  
  34.   #############################################################################
  35.   # TODO:                                                                     #
  36.   # Implement a vectorized version of the gradient for the structured SVM     #
  37.   # loss, storing the result in dW.                                           #
  38.   #                                                                           #
  39.   # Hint: Instead of computing the gradient from scratch, it may be easier    #
  40.   # to reuse some of the intermediate values that you used to compute the     #
  41.   # loss.                                                                     #
  42.   #############################################################################
  43.   margins[margins > 0] = 1
  44.   row_sum = np.sum(margins, axis=1)
  45.   margins[np.arange(num_train), y] = -row_sum
  46.   dW += np.dot(X.T, margins) / num_train + reg * W
  47.   #############################################################################
  48.   #                             END OF YOUR CODE                              #
  49.   #############################################################################
  50.  
  51.   return loss, dW

  4.     Train this linear classifier using stochastic gradient descent.
  6.     Inputs:
  7.     - X: A numpy array of shape (N, D) containing training data; there are N
  8.       training samples each of dimension D.
  9.     - y: A numpy array of shape (N,) containing training labels; y[i] = c
  10.       means that X[i] has label 0 <= c < C for C classes.
  11.     - learning_rate: (float) learning rate for optimization.
  12.     - reg: (float) regularization strength.

  17.     Outputs:
  18.     A list containing the value of the loss function at each training iteration.
  19.     """
  20.     num_train, dim = X.shape

  22.     if self.W is None:
  23.       # lazily initialize W
  24.       self.W = 0.001 * np.random.randn(dim, num_classes)
  25.  
  26.     # Run stochastic gradient descent to optimize W
  27.     loss_history = []
  28.     for it in range(num_iters):
  29.       X_batch = None
  30.       y_batch = None
  31.  
  32.       #########################################################################
  33.       # TODO:                                                                 #
  34.       # Sample batch_size elements from the training data and their           #
  35.       # corresponding labels to use in this round of gradient descent.        #
  36.       # Store the data in X_batch and their corresponding labels in           #
  37.       # y_batch; after sampling X_batch should have shape (dim, batch_size)   #

  12.     plt.axis('off')
  13.     plt.title(classes[i])
  14. plt.show()
