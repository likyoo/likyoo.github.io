实现过程 softmax_loss_naive 在文件 softmax.py 中

  1. def softmax_loss_naive(W, X, y, reg):
  2.   # Initialize the loss and gradient to zero.
  3.   loss = 0.0
  4.   dW = np.zeros_like(W)
  5.  
  6.   #############################################################################
  7.   # TODO: Compute the softmax loss and its gradient using explicit loops.     #
  8.   # Store the loss in loss and the gradient in dW. If you are not careful     #
  9.   # here, it is easy to run into numeric instability. Don't forget the        #
  10.   # regularization!                                                           #
  11.   #############################################################################
  12.   num_train = X.shape[0]
  13.   num_classes = W.shape[1]
  14.   for i in range(num_train):
  15.     # 算出score
  16.     f_i = X[i].dot(W)
  17.     # 为避免数据不稳定问题，每个分值向量都减去向量中的最大值
  18.     f_i -= np.max(f_i)
  19.     # 计算Loss
  20.     sum_j = np.sum(np.exp(f_i))
  21.     p = lambda k : np.exp(f_i[k]) / sum_j
  22.     loss += -np.log(p(y[i]))
  23.  
  24.     # 计算梯度
  25.     for k in range(num_classes):
  26.       p_k = p(k)
  27.       dW[:,k] += (p_k - (k == y[i])) * X[i]
  28.  
  29.   loss /= num_train
  30.   # 加上正则项
  31.   loss += 0.5 * reg * np.sum(W * W)
  32.   dW /= num_train
  33.   dW += reg * W
  34.  
  35.   #############################################################################
  36.   #                          END OF YOUR CODE                                 #
  37.   #############################################################################
  38.  
  39.   return loss, dW

softmax_loss_vectorized 在 softmax.py中实现

  1. def softmax_loss_vectorized(W, X, y, reg):
  2.   # Initialize the loss and gradient to zero.
  3.   loss = 0.0
  4.   dW = np.zeros_like(W)
  5.  
  6.   #############################################################################
  7.   # TODO: Compute the softmax loss and its gradient using no explicit loops.  #
  8.   # Store the loss in loss and the gradient in dW. If you are not careful     #
  9.   # here, it is easy to run into numeric instability. Don't forget the        #
  10.   # regularization!                                                           #
  11.   #############################################################################
  12.   num_train = X.shape[0]
  13.   f  = X.dot(W)
  14.   f -= np.max(f, axis=1, keepdims=True)
  15.   sum_f = np.sum(np.exp(f), axis=1, keepdims=True)
  16.   p = np.exp(f) / sum_f
  17.  
  18.   loss = np.sum(-np.log(p[np.arange(num_train), y]))
  19.  
  20.   ind = np.zeros_like(p)
  21.   ind[np.arange(num_train), y] = 1
  22.   dW = X.T.dot(p - ind)
  23.  
  24.   loss /= num_train
  25.   loss += 0.5 * reg * np.sum(W * W)
  26.   dW /= num_train
  27.   dW += reg * W
  28.   #############################################################################
  29.   #                          END OF YOUR CODE                                 #
  30.   #############################################################################
  31.  
  32.   return loss, dW
