通过交叉验证找到最好的k

  1. # 交叉验证
  2. num_folds = 5
  3. k_choices = [1, 3, 5, 8, 10, 12, 15, 20, 50, 100]
  4.  
  5. X_train_folds = []
  6. y_train_folds = []
  7.  
  8. X_train_folds = np.array_split(X_train, num_folds, axis=0)
  9. y_train_folds = np.array_split(y_train, num_folds, axis=0)
  10.  
  11.  
  12. # A dictionary holding the accuracies for different values of k that we find
  13. # when running cross-validation. After running cross-validation,
  14. # k_to_accuracies[k] should be a list of length num_folds giving the different
  15. # accuracy values that we found when using that value of k.
  16. k_to_accuracies = {}
  17.  
  18. for k in k_choices:
  19.  
  20.     k_to_accuracies[k] = []
  21.     for val_idx in range(num_folds):
  22.         X_val = X_train_folds[val_idx]
  23.         y_val = y_train_folds[val_idx]
  24.  
  25.         X_tra = X_train_folds[:val_idx] + X_train_folds[val_idx + 1:]
  26.         X_tra = np.reshape(X_tra, (X_train.shape[0] - X_val.shape[0], -1))
  27.         y_tra = y_train_folds[:val_idx] + y_train_folds[val_idx + 1:]
  28.         y_tra = np.reshape(y_tra, (X_train.shape[0] - X_val.shape[0],))
  29.  
  30.         classifier.train(X_tra, y_tra)
  31.         y_val_pre = classifier.predict(X_val, k, 0)
  32.  
  33.         right_arr = y_val_pre == y_val
  34.         accuracy = float(np.sum(right_arr)) / y_val.shape[0]
  35.         k_to_accuracies[k].append(accuracy)
  36.  
  37.  
  38. # Print out the computed accuracies
  39. for k in sorted(k_to_accuracies):
  40.     for accuracy in k_to_accuracies[k]:
  41.         print('k = %d, accuracy = %f' % (k, accuracy))

  11.     Train the classifier. For k-nearest neighbors this is just 
  12.     memorizing the training data.
  14.     Inputs:
  15.     - X: A numpy array of shape (num_train, D) containing the training data
  16.       consisting of num_train samples each of dimension D.
  17.     - y: A numpy array of shape (N,) containing the training labels, where
  18.          y[i] is the label for X[i].
  19.     """
  20.     self.X_train = X
  21.     self.y_train = y
  22.     
  23.   def predict(self, X, k=1, num_loops=0):
  24.     """
  25.     Predict labels for test data using this classifier.
  27.     Inputs:
  28.     - X: A numpy array of shape (num_test, D) containing test data consisting
  29.          of num_test samples each of dimension D.
  30.     - k: The number of nearest neighbors that vote for the predicted labels.
  31.     - num_loops: Determines which implementation to use to compute distances
  32.       between training points and testing points.
  34.     Returns:
  35.     - y: A numpy array of shape (num_test,) containing predicted labels for the
  36.       test data, where y[i] is the predicted label for the test point X[i].  
  37.     """
  38.     if num_loops == 0:
  39.       dists = self.compute_distances_no_loops(X)
  40.     elif num_loops == 1:
  41.       dists = self.compute_distances_one_loop(X)
  42.     elif num_loops == 2:
  43.       dists = self.compute_distances_two_loops(X)
  44.     else:
  45.       raise ValueError('Invalid value %d for num_loops' % num_loops)
  46.  
  47.     return self.predict_labels(dists, k=k)

  49.   def compute_distances_two_loops(self, X):

  61.       point.
  62.     """

  64.     num_train = self.X_train.shape[0]

  109.     num_test = X.shape[0]
