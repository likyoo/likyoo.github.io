leetcode 2 两数相加

2020-01-01 leetcode 0 评论

  3. /**
  4.  * Definition for singly-linked list.
  5.  * struct ListNode {
  6.  *     int val;
  7.  *     ListNode *next;
  8.  *     ListNode(int x) : val(x), next(NULL) {}
  9.  * };
  10.  */
  11. class Solution {
  12. public:
  13.     ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
  14.         int judge = 0, data = 0;
  15.         ListNode* L = new ListNode(-1);
  16.         ListNode* p = L;
  17.         while(l1||l2){
  18.             data = 0;
  19.             if(l1){
  20.                 data += l1->val;
  21.                 l1 = l1->next;
  22.             }
  23.             if(l2){
  24.                 data += l2->val;
  25.                 l2 = l2->next;
  26.             }
  27.  
  28.             data = data + judge;
  29.             p->next = new ListNode(data % 10);
  30.             p = p->next;
  31.             if(data >= 10)
  32.                 judge = 1;
  33.             else judge = 0;
  34.         }
  35.         if(judge == 1)
  36.             p->next = new ListNode(1);
  37.         return L->next;
  38.     }
  39. };
