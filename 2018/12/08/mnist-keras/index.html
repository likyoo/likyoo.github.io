mnist--keras

2018-12-08 machine learning 0 评论

1. activation function : sigmoid 

    loss function : mean squared error

    optimizer : SGD

  1. import numpy as np
  2. from keras.models import Sequential
  3. from keras.layers.core import Dense, Dropout, Activation
  4. from keras.layers import Conv2D, MaxPool2D, Flatten
  5. from keras.optimizers import SGD, Adam
  6. from keras.utils import np_utils
  7. from keras.datasets import mnist
  8.  
  9. def load_data():
  10.     (x_train, y_train), (x_test, y_test) = mnist.load_data()
  11.     number = 10000
  12.     x_train = x_train[0:number]
  13.     y_train = y_train[0:number]
  14.     x_train = x_train.reshape(number, 28*28)
  15.     x_test = x_test.reshape(x_test.shape[0], 28*28)
  16.     x_train = x_train.astype('float32')
  17.     x_test = x_test.astype('float32')
  18.  
  19.     y_train = np_utils.to_categorical(y_train, 10)
  20.     y_test = np_utils.to_categorical(y_test, 10)
  21.     x_train = x_train
  22.     x_test = x_test
  23.  
  24.     x_train = x_train / 255
  25.     x_test = x_test / 255
  26.     return (x_train, y_train), (x_test, y_test)
  27.  
  28. (x_train, y_train), (x_test, y_test) = load_data()
  29.  
  30. model = Sequential()
  31.  
  32. # add layer
  33. model.add(Dense(input_dim=28*28, units=733, activation='sigmoid'))
  34. model.add(Dense(units=733, activation='sigmoid'))
  35. model.add(Dense(units=733, activation='sigmoid'))
  36. model.add(Dense(units=10, activation='softmax'))
  37.  
  38. # select loss function; optimizer :决定 learning rate 的方法
  39. model.compile(loss='mse', optimizer=SGD(lr=0.1), metrics=['accuracy'])
  40.  
  41. # start training network
  42. model.fit(x_train, y_train, batch_size=100, epochs=20)
  43.  
  44. # 算出正确率
  45. train_result = model.evaluate(x_train, y_train)
  46. print('\n train Acc: ', train_result[1])
  47. result = model.evaluate(x_test, y_test)
  48. print('\n Test Acc: ', result[1])

结果：

    train Acc:  0.1127

    Test Acc:  0.1135

问题出在loss function上， mse确实不适合分类问题， 那将loss function换一下再试一下

8.那我们试一下dropout

  1. import numpy as np
  2. from keras.models import Sequential
  3. from keras.layers.core import Dense, Dropout, Activation
  4. from keras.layers import Conv2D, MaxPool2D, Flatten
  5. from keras.optimizers import SGD, Adam
  6. from keras.utils import np_utils
  7. from keras.datasets import mnist
  8.  
  9. def load_data():
  10.     (x_train, y_train), (x_test, y_test) = mnist.load_data()
  11.     number = 10000
  12.     x_train = x_train[0:number]
  13.     y_train = y_train[0:number]
  14.     x_train = x_train.reshape(number, 28*28)
  15.     x_test = x_test.reshape(x_test.shape[0], 28*28)
  16.     x_train = x_train.astype('float32')
  17.     x_test = x_test.astype('float32')
  18.  
  19.     y_train = np_utils.to_categorical(y_train, 10)
  20.     y_test = np_utils.to_categorical(y_test, 10)
  21.     x_train = x_train
  22.     x_test = x_test
  23.  
  24.     x_train = x_train / 255
  25.     x_test = x_test / 255
  26.  
  27.     # add noise
  28.     x_test = np.random.normal(x_test)
  29.     return (x_train, y_train), (x_test, y_test)
  30.  
  31. (x_train, y_train), (x_test, y_test) = load_data()
  32.  
  33. model = Sequential()
  34.  
  35. # add layer
  36. model.add(Dense(input_dim=28*28, units=733, activation='relu'))
  37. model.add(Dropout(0.7))
  38. model.add(Dense(units=733, activation='sigmoid'))
  39. model.add(Dropout(0.7))
  40. model.add(Dense(units=733, activation='sigmoid'))
  41. model.add(Dropout(0.7))
  42. model.add(Dense(units=10, activation='softmax'))
  43.  
  44.  
  45.  
  46. # select loss function; optimizer :决定 learning rate 的方法
  47. model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
  48.  
  49. # start training network
  50. model.fit(x_train, y_train, batch_size=100, epochs=20)
  51.  
  52. # 算出正确率
  53. train_result = model.evaluate(x_train, y_train)
  54. print('\n train Acc: ', train_result[1])
  55. result = model.evaluate(x_test, y_test)
  56. print('\n Test Acc: ', result[1])

结果：

     train Acc:  0.9923

    Test Acc:  0.5604

可见，training 的 performance 有所下降， 而testing 的 performance 是有所上升的
