HW1 PM2.5 代码解析

2018-11-15 machine learning 0 评论
  1. import csv
  2. import numpy as np
  3. from numpy.linalg import inv
  4. import random
  5. import math
  6. import sys
  7.  
  8. #读数据
  9. data = []
  10. # 每一个维度储存一种污染物的数据
  11. for i in range(18):
  12. 	data.append([])
  13.  
  14. n_row = 0
  15. text = open('C:/Users/liky/Desktop/PY/2017fall-ml-hw1-master/2017fall-ml-hw1-master/train.csv', 'r', encoding='big5')
  16. row = csv.reader(text , delimiter=",")
  17.  
  18. for r in row:
  19.     # 第0行沒有数据
  20.     if n_row != 0:
  21.         # 每一行只有第3-27格有值(1天內24小时的数值)
  22.         for i in range(3,27):
  23.             if r[i] != "NR":
  24.                 data[(n_row-1)%18].append(float(r[i]))
  25.             else:
  26.                 data[(n_row-1)%18].append(float(0))
  27.     n_row = n_row+1
  28. text.close()
  29.  
  30. #解析数据到(x,y)
  31. x = []
  32. y = []
  33. # 每12个月
  34. for i in range(12):
  35.     # 一個月取连续10小时的data可以有471组
  36.     for j in range(471):
  37.         x.append([])
  38.         # 18种污染物
  39.         for t in range(18):
  40.             # 连续9小時
  41.             for s in range(9):
  42.                 x[471*i+j].append(data[t][480*i+j+s] )   #一个x中的值为1到9时的值(18行9列)  对应的y的值为第10时的PM2.5的值
  43.         y.append(data[9][480*i+j+9])
  44.  
  45. #创建x和y的数组
  46. x = np.array(x)
  47. y = np.array(y)
  48.  
  49.  
  50. # add square term
  51. # x = np.concatenate((x,x**2), axis=1)
  52.  
  53. # add bias
  54. x = np.concatenate((np.ones((x.shape[0],1)),x), axis=1)  # bias
  55.  
  56. #初始参数
  57. w = np.zeros(len(x[0]))
  58. l_rate = 10
  59. repeat = 10000
  60.  
  61.  
  62. #check answer
  63. # use close form to check whether ur gradient descent is good
  64. # however, this cannot be used in hw1.sh
  65. # w = np.matmul(np.matmul(inv(np.matmul(x.transpose(),x)),x.transpose()),y)
  66.  
  67. #开始训练
  68. x_t = x.transpose()            #转置
  69. s_gra = np.zeros(len(x[0]))
  70.  
  71. for i in range(2):
  72.     hypo = np.dot(x,w)                      #矩阵相乘，得到一个行数和x相同，列数为1的矩阵
  73.     loss = hypo - y                         # hypo = w1*x1 + w2*x2 + ... + wt*xt + bias
  74.     cost = np.sum(loss**2) / len(x)         #...
  75.     cost_a = math.sqrt(cost)                #求loss值，对结果无影响
  76.     gra = np.dot(x_t,loss)                  # 求微分 loss = (hypo - y)   --->   &gra/&w1 = (hypo - y) * x1 = loss * x1
  77.     s_gra += gra**2                         #求 adagrad
  78.     ada = np.sqrt(s_gra)                    #...
  79.     w = w - l_rate * gra/ada                #得出下一步的点
  80.     print('iteration: %d | Cost: %f  ' % ( i,cost_a))
  81.  
  82.  
  83. # save model
  84. np.save('model.npy',w)
  85. # read model
  86. w = np.load('model.npy')
  87.  
  88. #读测试数据
  89. test_x = []
  90. n_row = 0
  91. text = open('C:/Users/liky/Desktop/PY/2017fall-ml-hw1-master/2017fall-ml-hw1-master/test.csv' ,"r")
  92. row = csv.reader(text , delimiter= ",")
  93.  
  94. for r in row:
  95.     if n_row %18 == 0:
  96.         test_x.append([])
  97.         for i in range(2,11):
  98.             test_x[n_row//18].append(float(r[i]) )           #把每天的九个小时的数据作为一组存到一个test_x里
  99.     else :
  100.         for i in range(2,11):
  101.             if r[i] !="NR":
  102.                 test_x[n_row//18].append(float(r[i]))
  103.             else:
  104.                 test_x[n_row//18].append(0)
  105.     n_row = n_row+1
  106. text.close()
  107. test_x = np.array(test_x)
  108.  
  109. # add square term
  110. # test_x = np.concatenate((test_x,test_x**2), axis=1)
  111.  
  112. # add bias
  113. test_x = np.concatenate((np.ones((test_x.shape[0],1)),test_x), axis=1)
  114.  
  115. #得到结果
  116. ans = []
  117. for i in range(len(test_x)):
  118.     ans.append(["id_"+str(i)])
  119.     a = np.dot(w,test_x[i])
  120.     ans[i].append(a)
  121.  
  122.  
  123.  
  124. filename = "C:/Users/liky/Desktop/PY/2017fall-ml-hw1-master/2017fall-ml-hw1-master/predict.csv"
  125. text = open(filename, "w+")
  126. s = csv.writer(text,delimiter=',',lineterminator='\n')
  127. s.writerow(["id","value"])
  128. for i in range(len(ans)):
  129.     s.writerow(ans[i])
  130. text.close()
