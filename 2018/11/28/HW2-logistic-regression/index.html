HW2 Logistic Regression 代码解析

  1. import os, sys
  2. import numpy as np
  3. from random import shuffle

  8. def load_data(train_data_path, train_label_path, test_data_path):
  9.     X_train = pd.read_csv(train_data_path, sep=',', header=0)
  10.     X_train = np.array(X_train.values)
  11.     Y_train = pd.read_csv(train_label_path, sep=',', header=0)
  12.     Y_train = np.array(Y_train.values)
  13.     X_test = pd.read_csv(test_data_path, sep=',', header=0)
  14.     X_test = np.array(X_test.values)
  15.  
  16.     return (X_train, Y_train, X_test)
  17.  
  18. def _shuffle(X, Y):
  19.     randomize = np.arange(len(X))
  20.     np.random.shuffle(randomize)
  21.     return (X[randomize], Y[randomize])
  22.  
  23. # 数据标准化， 使数据便于处理
  24. # 做特征缩放 : Feature Scaling
  25. def normalize(X_all, X_test):
  26.     # Feature normalization with train and test X
  27.     X_train_test = np.concatenate((X_all, X_test))              # 先做拼接处理
  28.     print(X_train_test.shape)
  29.     mu = (sum(X_train_test) / X_train_test.shape[0])
  30.     sigma = np.std(X_train_test, axis=0)                        # 计算标准差
  31.     # 将mu sigma处理（复制）成可以直接操作的数组
  32.     mu = np.tile(mu, (X_train_test.shape[0], 1))
  33.     sigma = np.tile(sigma, (X_train_test.shape[0], 1))
  34.     X_train_test_normed = (X_train_test - mu) / sigma
  35.  
  36.     # Split to train, test again
  37.     X_all = X_train_test_normed[0:X_all.shape[0]]
  38.     X_test = X_train_test_normed[X_all.shape[0]:]
  39.     return X_all, X_test
  40.  
  41. def split_valid_set(X_all, Y_all, percentage):
  42.     all_data_size = len(X_all)
  43.     valid_data_size = int(floor(all_data_size * percentage))
  44.  
  45.     X_all, Y_all = _shuffle(X_all, Y_all)     #打乱顺序
  46.  
  47.     X_train, Y_train = X_all[0:valid_data_size], Y_all[0:valid_data_size]
  48.     X_valid, Y_valid = X_all[valid_data_size:], Y_all[valid_data_size:]
  49.  
  50.     return X_train, Y_train, X_valid, Y_valid
  51.  
  52. def sigmoid(z):
  53.     res = 1 / (1.0 + np.exp(-z))
  54.     return np.clip(res, 1e-8, 1-(1e-8))
  55.  
  56. def valid(w, b, X_valid, Y_valid):
  57.     valid_data_size = len(X_valid)
  58.  
  59.     z = (np.dot(X_valid, np.transpose(w)) + b)
  60.     y = sigmoid(z)
  61.     y_ = np.around(y)
  62.     result = (np.squeeze(Y_valid) == y_)
  63.     print('Validation acc = %f' % (float(result.sum()) / valid_data_size))
  64.     return
  65.  
  66. # train的目的是得到 w, b 参数
  67. def train(X_all, Y_all, save_dir):
  68.     pass
  69.     # Split a 10%-validation set from the training set
  70.     # ？？？不懂为什么只取10%作为training data
  71.     valid_set_percentage = 0.1
  72.     X_train, Y_train, X_valid, Y_valid = split_valid_set(X_all, Y_all, valid_set_percentage)   #把X_train和Y_train分别分成十份，取10%作为training set
  73.  
  74.     # Initiallize parameter, hyperparameter
  75.     w = np.zeros((106,))
  76.     b = np.zeros((1,))
  77.     l_rate = 0.1
  78.     batch_size = 32
  79.     train_data_size = len(X_train)
  80.     step_num = int(floor(train_data_size / batch_size))
  81.     epoch_num = 1000
  82.     save_param_iter = 50
  83.  
  84.     # Start training
  85.     total_loss = 0.0
  86.     for epoch in range(1, epoch_num):
  87.         # Do validation and parameter saving
  88.         # 每epoch_num个epoch保存一次
  89.         if (epoch) % save_param_iter == 0:
  90.             print('=====Saving Param at epoch %d=====' % epoch)
  91.             if not os.path.exists(save_dir):
  92.                 os.mkdir(save_dir)
  93.             np.savetxt(os.path.join(save_dir, 'w'), w)
  94.             np.savetxt(os.path.join(save_dir, 'b'), [b,])
  95.  
  96.             # 打印 trianing set 的loss
  97.             print('epoch avg loss = %f' % (total_loss / (float(save_param_iter) * train_data_size)))
  98.             total_loss = 0.0
  99.  
  100.             # 打印valid中的正确率
  101.             valid(w, b, X_valid, Y_valid)
  102.  
  103.         # Random shuffle
  104.         X_train, Y_train = _shuffle(X_train, Y_train)
  105.  
  106.         # Train with batch
  107.         for idx in range(step_num):
  108.             # 取出一个batch
  109.             X = X_train[idx*batch_size:(idx+1)*batch_size]
  110.             Y = Y_train[idx*batch_size:(idx+1)*batch_size]
  111.  
  112.             # logistic过程
  113.             z = np.dot(X, np.transpose(w)) + b
  114.             y = sigmoid(z)
  115.             # loss function : cross entropy
  116.             cross_entropy = -1 * (np.dot(np.squeeze(Y), np.log(y)) + np.dot((1 - np.squeeze(Y)), np.log(1 - y)))
  117.             total_loss += cross_entropy
  118.  
  119.             w_grad = np.mean(-1 * X * (np.squeeze(Y) - y).reshape((batch_size,1)), axis=0)
  120.             b_grad = np.mean(-1 * (np.squeeze(Y) - y))
  121.  
  122.             # SGD updating parameters
  123.             # 用随机梯度递减法， 不断更新
  124.             w = w - l_rate * w_grad
  125.             b = b - l_rate * b_grad
  126.  
  127.     return
  128.  
  129. def infer(X_test, save_dir, output_dir):
  130.     test_data_size = len(X_test)
  131.  
  132.     # Load parameters
  133.     print('=====Loading Param from %s=====' % save_dir)
  134.     w = np.loadtxt(os.path.join(save_dir, 'w'))
  135.     b = np.loadtxt(os.path.join(save_dir, 'b'))
  136.  
  137.     # predict
  138.     z = (np.dot(X_test, np.transpose(w)) + b)
  139.     y = sigmoid(z)
  140.     y_ = np.around(y)
  141.  
  142.     print('=====Write output to %s =====' % output_dir)
  143.     if not os.path.exists(output_dir):
  144.         os.mkdir(output_dir)
  145.     output_path = os.path.join(output_dir, 'log_prediction.csv')
  146.     with open(output_path, 'w') as f:
  147.         f.write('id,label\n')
  148.         for i, v in  enumerate(y_):
  149.             f.write('%d,%d\n' %(i+1, v))
  150.  
  151.     return

  154.     # Load feature and label

  165.     else:
  166.         print("Error: Argument --train or --infer not found")
